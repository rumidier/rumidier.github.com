날짜 관련 예제

#!/usr/bin/env perl 

use 5.010;
use strict;
use warnings;
use POSIX qw(strftime);
use DateTime;

say "POSOX";
say POSIX::strftime('%Y-%m-%d %H:%M:%S', localtime);

say "DateTime";
say DateTime->now( time_zone => 'local' )->strftime('%Y-%m-%d %H:%M:%S');
